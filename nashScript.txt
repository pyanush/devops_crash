#!/bin/bash

sftp -v testuser@207.244.229.74 <<EOF
lcd /root/
cd /home/testuser/
put 1.txt
quit
EOF
SFTP_RETURN_CODE=${?}
# If the return code is non-zero then the upload was not successful
if [[ 0 != ${SFTP_RETURN_CODE} ]]
   then
   echo "Natalia upload for 1.txt failed"
   exit ${SFTP_RETURN_CODE}
else
   echo "Natalia upload for 1.txt was successful"
fi
exit 0
